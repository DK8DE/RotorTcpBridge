pyinstaller --noconfirm --clean RotorTcpBridge.spec

.\build.ps1
# Version auf 1.3 setzen, bauen, Installer erstellen

.\build.ps1 -Version "1.3" -SkipInstaller

version.py  (APP_VERSION = "1.3")
